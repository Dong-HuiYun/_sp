# 第六章：網路程式設計

## 6.1 網路基礎概念

網路程式設計是系統程式設計的重要組成部分，涉及到作業系統如何處理網路通訊。

### 6.1.1 OSI 參考模型

```
┌─────────────────────────────────────┐
│           第七層：應用層             │ ← HTTP, FTP, SMTP
├─────────────────────────────────────┤
│           第六層：表示層             │ ← SSL/TLS, JPEG
├─────────────────────────────────────┤
│           第五層：會話層             │ ← NetBIOS, RPC
├─────────────────────────────────────┤
│           第四層：傳輸層             │ ← TCP, UDP
├─────────────────────────────────────┤
│           第三層：網路層             │ ← IP, ICMP
├─────────────────────────────────────┤
│           第二層：資料連結層        │ ← Ethernet, PPP
├─────────────────────────────────────┤
│           第一層：實體層             │ ← 光纖, 銅線
└─────────────────────────────────────┘
```

```
┌─────────────────────────────────────┐
│         應用層 (Application)         │ HTTP, FTP, DNS

│         網際層 (Internet)           │ IP, ICMP, ARP
```

## 6.2 Socket 程式設計

### 6.2.1 Socket 概述

Socket 是網路通訊的端點，用於行程之間的網路通訊。

```c
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

typedef enum {
    SOCKET_STREAM,     // TCP (面向連接)
    SOCKET_DGRAM,      // UDP (無連接)
    SOCKET_RAW         // 原始 Socket
} SocketType;

// 創建 Socket
int create_socket(SocketType type) {
    int domain = AF_INET;      // IPv4
    int socket_type;
    
    switch (type) {
        case SOCKET_STREAM:
            socket_type = SOCK_STREAM;
            break;
        case SOCKET_DGRAM:
            socket_type = SOCK_DGRAM;
            break;
        case SOCKET_RAW:
            socket_type = SOCK_RAW;
            break;
    }
    
    int fd = socket(domain, socket_type, 0);
    if (fd == -1) {
        perror("socket");
        return -1;
    }
    
    return fd;
}
```

### 6.2.2 TCP 客戶端

```c
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define SERVER_IP "127.0.0.1"
#define SERVER_PORT 8080
#define BUFFER_SIZE 1024

int tcp_client_example(void) {
    int sockfd;
    struct sockaddr_in server_addr;
    char buffer[BUFFER_SIZE];
    
    sockfd = socket(AF_INET, SOCK_STREAM, 0);
    if (sockfd == -1) {
        perror("socket");
        return -1;
    }
    
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(SERVER_PORT);
    
    if (inet_pton(AF_INET, SERVER_IP, &server_addr.sin_addr) <= 0) {
        perror("inet_pton");
        close(sockfd);
        return -1;
    }
    
    if (connect(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr)) == -1) {
        perror("connect");
        close(sockfd);
        return -1;
    }
    
    printf("Connected to server %s:%d\n", SERVER_IP, SERVER_PORT);
    
    const char *message = "Hello from client!";
    send(sockfd, message, strlen(message), 0);
    
    ssize_t bytes_received = recv(sockfd, buffer, BUFFER_SIZE - 1, 0);
    if (bytes_received > 0) {
        buffer[bytes_received] = '\0';
        printf("Received from server: %s\n", buffer);
    }
    
    close(sockfd);
    
    return 0;
}
```

### 6.2.3 TCP 伺服器

```c
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <signal.h>

#define PORT 8080
#define BUFFER_SIZE 1024
#define BACKLOG 10

int server_socket;

void signal_handler(int sig) {
    close(server_socket);
    exit(0);
}

int tcp_server_example(void) {
    int clientfd;
    struct sockaddr_in server_addr, client_addr;
    socklen_t client_len;
    char buffer[BUFFER_SIZE];
    
    signal(SIGINT, signal_handler);
    
    server_socket = socket(AF_INET, SOCK_STREAM, 0);
    if (server_socket == -1) {
        perror("socket");
        return -1;
    }
    
    int opt = 1;
    setsockopt(server_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
    
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);
    
    if (bind(server_socket, (struct sockaddr*)&server_addr, sizeof(server_addr)) == -1) {
        perror("bind");
        close(server_socket);
        return -1;
    }
    
    if (listen(server_socket, BACKLOG) == -1) {
        perror("listen");
        close(server_socket);
        return -1;
    }
    
    printf("TCP Server listening on port %d...\n", PORT);
    
    while (1) {
        client_len = sizeof(client_addr);
        clientfd = accept(server_socket, (struct sockaddr*)&client_addr, &client_len);
        
        if (clientfd == -1) {
            perror("accept");
            continue;
        }
        
        printf("Client connected: %s:%d\n",
               inet_ntoa(client_addr.sin_addr),
               ntohs(client_addr.sin_port));
        
        pid_t pid = fork();
        if (pid == 0) {
            close(server_socket);
            
            ssize_t bytes_received = recv(clientfd, buffer, BUFFER_SIZE - 1, 0);
            if (bytes_received > 0) {
                buffer[bytes_received] = '\0';
                printf("Received: %s\n", buffer);
                
                const char *response = "Message received!";
                send(clientfd, response, strlen(response), 0);
            }
            
            close(clientfd);
            exit(0);
        } else if (pid > 0) {
            close(clientfd);
        }
    }
    
    close(server_socket);
    
    return 0;
}
```

## 6.4 多客戶端處理

### 6.4.1 select 多路複用

```c
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_CLIENTS 1024
#define PORT 8080
#define BUFFER_SIZE 1024

int tcp_multiplex_server(void) {
    int master_socket, client_socket[MAX_CLIENTS];
    struct sockaddr_in address;
    fd_set readfds;
    int max_sd, sd, activity;
    char buffer[BUFFER_SIZE];
    
    for (int i = 0; i < MAX_CLIENTS; i++) {
        client_socket[i] = 0;
    }
    
    master_socket = socket(AF_INET, SOCK_STREAM, 0);
    if (master_socket == -1) {
        perror("socket");
        return -1;
    }
    
    int opt = 1;
    setsockopt(master_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
    
    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);
    
    bind(master_socket, (struct sockaddr*)&address, sizeof(address));
    listen(master_socket, 5);
    
    printf("Server listening on port %d...\n", PORT);
    
    while (1) {
        FD_ZERO(&readfds);
        FD_SET(master_socket, &readfds);
        max_sd = master_socket;
        
        for (int i = 0; i < MAX_CLIENTS; i++) {
            sd = client_socket[i];
            if (sd > 0) {
                FD_SET(sd, &readfds);
            }
            if (sd > max_sd) {
                max_sd = sd;
            }
        }
        
        activity = select(max_sd + 1, &readfds, NULL, NULL, NULL);
        
        if (activity < 0) {
            perror("select");
            continue;
        }
        
        if (FD_ISSET(master_socket, &readfds)) {
            struct sockaddr_in client_addr;
            socklen_t addr_len = sizeof(client_addr);
            int new_socket = accept(master_socket, 
                                    (struct sockaddr*)&client_addr, &addr_len);
            
            printf("New connection: socket %d\n", new_socket);
            
            for (int i = 0; i < MAX_CLIENTS; i++) {
                if (client_socket[i] == 0) {
                    client_socket[i] = new_socket;
                    break;
                }
            }
        }
        
        for (int i = 0; i < MAX_CLIENTS; i++) {
            sd = client_socket[i];
            
            if (FD_ISSET(sd, &readfds)) {
                ssize_t bytes_received = recv(sd, buffer, BUFFER_SIZE, 0);
                
                if (bytes_received <= 0) {
                    printf("Client disconnected\n");
                    close(sd);
                    client_socket[i] = 0;
                } else {
                    buffer[bytes_received] = '\0';
                    printf("Received from socket %d: %s\n", sd, buffer);
                    
                    send(sd, buffer, bytes_received, 0);
                }
            }
        }
    }
    
    close(master_socket);
    return 0;
}
```

### 6.4.2 poll 多路複用

```c
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <poll.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

#define PORT 8080
#define MAX_CLIENTS 1024
#define BUFFER_SIZE 1024

int poll_server_example(void) {
    int listen_socket;
    struct sockaddr_in server_addr;
    struct pollfd fds[MAX_CLIENTS + 1];
    int nfds = 1;
    
    listen_socket = socket(AF_INET, SOCK_STREAM, 0);
    if (listen_socket == -1) {
        perror("socket");
        return -1;
    }
    
    int opt = 1;
    setsockopt(listen_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
    
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);
    
    bind(listen_socket, (struct sockaddr*)&server_addr, sizeof(server_addr));
    listen(listen_socket, 5);
    
    fds[0].fd = listen_socket;
    fds[0].events = POLLIN;
    
    for (int i = 1; i <= MAX_CLIENTS; i++) {
        fds[i].fd = -1;
    }
    
    printf("Server listening on port %d...\n", PORT);
    
    while (1) {
        int activity = poll(fds, nfds, -1);
        
        if (activity < 0) {
            perror("poll");
            continue;
        }
        
        if (fds[0].revents & POLLIN) {
            struct sockaddr_in client_addr;
            socklen_t addr_len = sizeof(client_addr);
            int client_socket = accept(listen_socket, 
                                      (struct sockaddr*)&client_addr, &addr_len);
            
            for (int i = 1; i <= MAX_CLIENTS; i++) {
                if (fds[i].fd == -1) {
                    fds[i].fd = client_socket;
                    fds[i].events = POLLIN;
                    nfds = (i >= nfds) ? i + 1 : nfds;
                    break;
                }
            }
        }
        
        for (int i = 1; i < nfds; i++) {
            if (fds[i].fd == -1) continue;
            
            if (fds[i].revents & (POLLIN | POLLHUP)) {
                char buffer[BUFFER_SIZE];
                ssize_t bytes_received = recv(fds[i].fd, buffer, BUFFER_SIZE, 0);
                
                if (bytes_received <= 0) {
                    close(fds[i].fd);
                    fds[i].fd = -1;
                } else {
                    buffer[bytes_received] = '\0';
                    send(fds[i].fd, buffer, bytes_received, 0);
                }
            }
        }
    }
    
    close(listen_socket);
    return 0;
}
```

### 6.4.3 epoll（Linux 特有）

```c
#include <sys/epoll.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define PORT 8080
#define MAX_EVENTS 1024
#define BUFFER_SIZE 1024

int epoll_server_example(void) {
    int listen_socket, epoll_fd;
    struct epoll_event ev, events[MAX_EVENTS];
    struct sockaddr_in server_addr;
    
    listen_socket = socket(AF_INET, SOCK_STREAM, 0);
    if (listen_socket == -1) {
        perror("socket");
        return -1;
    }
    
    int opt = 1;
    setsockopt(listen_socket, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
    
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);
    
    bind(listen_socket, (struct sockaddr*)&server_addr, sizeof(server_addr));
    listen(listen_socket, 5);
    
    epoll_fd = epoll_create1(0);
    if (epoll_fd == -1) {
        perror("epoll_create1");
        close(listen_socket);
        return -1;
    }
    
    ev.events = EPOLLIN;
    ev.data.fd = listen_socket;
    
    if (epoll_ctl(epoll_fd, EPOLL_CTL_ADD, listen_socket, &ev) == -1) {
        perror("epoll_ctl");
        close(listen_socket);
        close(epoll_fd);
        return -1;
    }
    
    printf("Server listening on port %d...\n", PORT);
    
    while (1) {
        int nfds = epoll_wait(epoll_fd, events, MAX_EVENTS, -1);
        
        if (nfds == -1) {
            perror("epoll_wait");
            continue;
        }
        
        for (int i = 0; i < nfds; i++) {
            if (events[i].data.fd == listen_socket) {
                struct sockaddr_in client_addr;
                socklen_t addr_len = sizeof(client_addr);
                int client_socket = accept(listen_socket, 
                                          (struct sockaddr*)&client_addr, &addr_len);
                
                ev.events = EPOLLIN | EPOLLET;
                ev.data.fd = client_socket;
                
                epoll_ctl(epoll_fd, EPOLL_CTL_ADD, client_socket, &ev);
            } else {
                char buffer[BUFFER_SIZE];
                ssize_t bytes_received = recv(events[i].data.fd, buffer, BUFFER_SIZE, 0);
                
                if (bytes_received <= 0) {
                    close(events[i].data.fd);
                } else {
                    send(events[i].data.fd, buffer, bytes_received, 0);
                }
            }
        }
    }
    
    close(listen_socket);
    close(epoll_fd);
    return 0;
}
```

## 6.5 位址轉換函式

### 6.5.1 IP 位址轉換

```c
#include <arpa/inet.h>
#include <stdio.h>
#include <string.h>

int ip_conversion_examples(void) {
    struct in_addr addr;
    struct in6_addr addr6;
    char ip_str[INET_ADDRSTRLEN];
    char ip_str6[INET6_ADDRSTRLEN];
    
    // 文字轉二進制（IPv4）
    if (inet_pton(AF_INET, "192.168.1.1", &addr) == 1) {
        printf("IPv4 address converted: %u\n", ntohl(addr.s_addr));
    }
    
    // 二進制轉文字（IPv4）
    if (inet_ntop(AF_INET, &addr, ip_str, INET_ADDRSTRLEN) != NULL) {
        printf("IPv4 address: %s\n", ip_str);
    }
    
    // IPv6 轉換
    if (inet_pton(AF_INET6, "2001:0db8:85a3:0000:0000:8a2e:0370:7334", &addr6) == 1) {
        printf("IPv6 address converted successfully\n");
    }
    
    if (inet_ntop(AF_INET6, &addr6, ip_str6, INET6_ADDRSTRLEN) != NULL) {
        printf("IPv6 address: %s\n", ip_str6);
    }
    
    return 0;
}
```

```c
int hostname_resolution(void) {
    struct hostent *host;
    struct in_addr **addr_list;

    if (host == NULL) {
        herror("gethostbyname");
        return -1;
    }

    printf("Aliases:\n");

        printf("  %s\n", *alias);

    printf("Address type: %s\n",

    for (int i = 0; addr_list[i] != NULL; i++) {
        printf("  %s\n", inet_ntoa(*addr_list[i]));
    }
    
    return 0;
}

int get_local_addresses(void) {

        herror("gethostbyname");
```

```c
    }
    
    int broadcast_enable = 1;
    setsockopt(sockfd, SOL_SOCKET, SO_BROADCAST, 
               &broadcast_enable, sizeof(broadcast_enable));
    
    memset(&broadcast_addr, 0, sizeof(broadcast_addr));
    broadcast_addr.sin_family = AF_INET;
    broadcast_addr.sin_port = htons(8080);
    broadcast_addr.sin_addr.s_addr = inet_addr("255.255.255.255");
    
    sendto(sockfd, message, strlen(message), 0,
           (struct sockaddr*)&broadcast_addr, sizeof(broadcast_addr));
    
    printf("Broadcast message sent\n");
    
    close(sockfd);
    return 0;
}
```

### 6.6.2 多播

```c
#include <sys/types.h>

#include <arpa/inet.h>
#include <unistd.h>

#include <string.h>

#define MULTICAST_GROUP "239.255.255.250"
#define PORT 8080

int multicast_sender(void) {
    int sockfd;
    struct sockaddr_in multicast_addr;
    
    sockfd = socket(AF_INET, SOCK_DGRAM, 0);
    if (sockfd == -1) {
        return -1;
    }
    
    memset(&multicast_addr, 0, sizeof(multicast_addr));
    multicast_addr.sin_family = AF_INET;
    multicast_addr.sin_port = htons(PORT);
    multicast_addr.sin_addr.s_addr = inet_addr(MULTICAST_GROUP);
    
    const char *message = "Multicast message!";
    sendto(sockfd, message, strlen(message), 0,
           (struct sockaddr*)&multicast_addr, sizeof(multicast_addr));
    
    printf("Multicast message sent to %s:%d\n", MULTICAST_GROUP, PORT);
    
    close(sockfd);
    return 0;
}

int multicast_receiver(void) {
    int sockfd;
    struct sockaddr_in local_addr, multicast_addr;
    char buffer[1024];
    
    sockfd = socket(AF_INET, SOCK_DGRAM, 0);
    if (sockfd == -1) {
        return -1;
    }
    
    memset(&local_addr, 0, sizeof(local_addr));
    local_addr.sin_family = AF_INET;
    local_addr.sin_addr.s_addr = INADDR_ANY;
    local_addr.sin_port = htons(PORT);
    
    bind(sockfd, (struct sockaddr*)&local_addr, sizeof(local_addr));
    
    struct ip_mreq mreq;
    mreq.imr_multiaddr.s_addr = inet_addr(MULTICAST_GROUP);
    mreq.imr_interface.s_addr = INADDR_ANY;
    
    setsockopt(sockfd, IPPROTO_IP, IP_ADD_MEMBERSHIP, &mreq, sizeof(mreq));
    
    printf("Waiting for multicast messages...\n");
    
    struct sockaddr_in sender_addr;
    socklen_t sender_len = sizeof(sender_addr);
    
    ssize_t bytes_received = recvfrom(sockfd, buffer, sizeof(buffer) - 1, 0,
                                      (struct sockaddr*)&sender_addr, &sender_len);
    if (bytes_received > 0) {
        buffer[bytes_received] = '\0';
        printf("Received: %s\n", buffer);
    }
    
    close(sockfd);
    return 0;
}
```

### 6.6.3 Unix Domain Socket

```c
#include <sys/types.h>
#include <sys/socket.h>
#include <sys/un.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

#define SOCKET_PATH "/tmp/my_socket"

int unix_domain_server(void) {
    int server_fd, client_fd;
    struct sockaddr_un server_addr, client_addr;
    socklen_t client_len;
    char buffer[256];
    
    unlink(SOCKET_PATH);
    
    server_fd = socket(AF_UNIX, SOCK_STREAM, 0);
    if (server_fd == -1) {
        perror("socket");
        return -1;
    }
    
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sun_family = AF_UNIX;
    strncpy(server_addr.sun_path, SOCKET_PATH, sizeof(server_addr.sun_path) - 1);
    
    bind(server_fd, (struct sockaddr*)&server_addr, sizeof(server_addr));
    listen(server_fd, 5);
    
    printf("Unix domain server listening on %s\n", SOCKET_PATH);
    
    client_len = sizeof(client_addr);
    client_fd = accept(server_fd, (struct sockaddr*)&client_addr, &client_len);
    
    ssize_t bytes_received = recv(client_fd, buffer, sizeof(buffer) - 1, 0);
    if (bytes_received > 0) {
        buffer[bytes_received] = '\0';
        printf("Received: %s\n", buffer);
    }
    
    close(client_fd);
    close(server_fd);
    unlink(SOCKET_PATH);
    
    return 0;
}

int unix_domain_client(void) {
    int sockfd;
    struct sockaddr_un server_addr;
    char buffer[256];
    
    sockfd = socket(AF_UNIX, SOCK_STREAM, 0);
    if (sockfd == -1) {
        perror("socket");
        return -1;
    }
    
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sun_family = AF_UNIX;
    strncpy(server_addr.sun_path, SOCKET_PATH, sizeof(server_addr.sun_path) - 1);
    
    if (connect(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr)) == -1) {
        perror("connect");
        close(sockfd);
        return -1;
    }
    
    const char *message = "Hello from Unix domain client!";
    send(sockfd, message, strlen(message), 0);
    
    printf("Message sent\n");
    
    close(sockfd);
    return 0;
}
```
