# 第三章：記憶體管理

## 3.1 記憶體層次結構

現代電腦系統的記憶體可以分為多個層次，從最快但最小的暫存器到最慢但最大的輔助儲存。

```
┌─────────────────────────────────────┐
│  CPU 暫存器（Registers）            │ ← 最快、最小
├─────────────────────────────────────┤
│  L1 快取（Cache）                   │
├─────────────────────────────────────┤
│  L2 快取（Cache）                   │
├─────────────────────────────────────┤
│  L3 快取（Cache）                   │
├─────────────────────────────────────┤
│  主記憶體（RAM）                    │
├─────────────────────────────────────┤
│  固態硬碟（SSD）                    │
├─────────────────────────────────────┤
│  傳統硬碟（HDD）                    │ ← 最慢、最大
└─────────────────────────────────────┘
```

### 3.1.1 暫存器

暫存器是CPU內部的最快儲存單元，用於儲存指令、資料和位址。

```c
// x86-64 主要暫存器
// 通用暫存器
// rax, rbx, rcx, rdx, rsi, rdi, rbp, rsp, r8-r15

// 指標暫存器
// rip: 指令指標
// rsp: 堆疊指標
// rbp: 基底指標

// 在C語言中內聯組譯直接使用暫存器
__asm__ volatile (
    "movq $10, %rax\n\t"
    "addq $5, %rax\n\t"
    "movq %rax, -8(%rbp)"
);
```

### 3.1.2 快取記憶體

快取記憶體位於CPU和主記憶體之間，用於減少記憶體存取時間。

```c
// 簡化的高速緩存模擬
typedef struct {
    unsigned long tag;
    unsigned long index;
    unsigned long offset;
    int valid;
    char data[64];
} CacheLine;

typedef struct {
    CacheLine *lines;
    int size;
    int associativity;
    int block_size;
} Cache;

Cache* cache_create(int size, int associativity, int block_size) {
    Cache *cache = (Cache*)malloc(sizeof(Cache));
    cache->size = size;
    cache->associativity = associativity;
    cache->block_size = block_size;
    
    int num_sets = size / (associativity * block_size);
    cache->lines = (CacheLine*)calloc(num_sets * associativity, sizeof(CacheLine));
    
    return cache;
}

int cache_access(Cache *cache, unsigned long address) {
    int block_offset_bits = __builtin_ctz(cache->block_size);
    int index_bits = __builtin_ctz(cache->size / (cache->associativity * cache->block_size));
    
    unsigned long offset = address & ((1 << block_offset_bits) - 1);
    unsigned long index = (address >> block_offset_bits) & ((1 << index_bits) - 1);
    unsigned long tag = address >> (block_offset_bits + index_bits);
    
    int num_sets = cache->size / (cache->associativity * cache->block_size);
    
    for (int i = 0; i < cache->associativity; i++) {
        CacheLine *line = &cache->lines[index * cache->associativity + i];
        if (line->valid && line->tag == tag) {
            return 1;
        }
    }
    
    return 0;
}
```

### 3.1.3 主記憶體

主記憶體（RAM）用於儲存正在執行的程式和資料。

```c
// 記憶體配置範例
#include <stdlib.h>

int main() {
    int *array = (int*)malloc(sizeof(int) * 1000);
    
    for (int i = 0; i < 1000; i++) {
        array[i] = i * i;
    }
    
    free(array);
    return 0;
}
```

## 3.2 記憶體位址空間

### 3.2.1 實體位址與虛擬位址

實體位址是實際硬體記憶體的位置，而虛擬位址是程式使用的邏輯位址。

```
虛擬位址空間                    實體記憶體
┌─────────────┐              ┌─────────────┐
│   0x0000    │              │   0x0000    │
│     ↓       │              │     ↓       │
│  程式使用   │  ──────────→ │  實體RAM    │
│  虛擬位址   │   記憶體     │  實體位址   │
│             │   管理單元   │             │
│   0xFFFF    │  (MMU)      │             │
└─────────────┘              └─────────────┘
```

### 3.2.2 行程的虛擬位址空間

在現代作業系統中，每個行程都有其獨立的虛擬位址空間。

```
Linux 行程虛擬位址空間佈局（32位元）：

0xFFFFFFFF ─┬─────────────────────┐
            │     核心空間        │ 1GB
0xC0000000 ─┼─────────────────────┤
            │       堆疊          │ 向高址增長
            │        ↓            │
            │                     │
            │     未使用          │
            │                     │
            │        ↑            │
            │       堆積          │ 向低址增長
0x08048000 ─┼─────────────────────┤
            │     BSS區段         │ 未初始化全域變數
            │     資料區段        │ 初始化全域變數
            │     文字區段         │ 程式碼
0x00000000 ─┴─────────────────────┘
```

## 3.3 分頁機制

分頁是現代作業系統使用的一種虛擬記憶體管理技術。

### 3.3.1 分頁表結構

```c
// 簡化的分頁表項結構
typedef struct {
    unsigned int present : 1;        // 在記憶體中
    unsigned int rw : 1;            // 讀寫許可權
    unsigned int user : 1;          // 使用者/核心模式
    unsigned int accessed : 1;      // 被訪問過
    unsigned int dirty : 1;        // 被修改過
    unsigned int unused : 7;       // 未使用
    unsigned int frame : 20;        // 實體頁框號
} PageTableEntry;

// 分頁轉換
typedef struct {
    unsigned int page_number;
    unsigned int offset;
} VirtualAddress;

VirtualAddress translate_virtual_address(void *virtual_addr, int page_size) {
    VirtualAddress va;
    unsigned long addr = (unsigned long)virtual_addr;
    
    va.page_number = addr / page_size;
    va.offset = addr % page_size;
    
    return va;
}

unsigned long translate_to_physical(PageTableEntry *page_table, 
                                     VirtualAddress va,
                                     int page_size) {
    if (!page_table[va.page_number].present) {
        fprintf(stderr, "Page fault!\n");
        return 0;
    }
    
    unsigned long frame = page_table[va.page_number].frame;
    return (frame * page_size) + va.offset;
}
```

### 3.3.2 多級分頁

大型位址空間需要多級分頁表來節省記憶體。

```c
// 兩級分頁結構
typedef struct {
    PageTableEntry entries[1024];
} PageDirectory;

typedef struct {
    PageTableEntry entries[1024];
} PageTable;

// x86 兩級分頁轉換
unsigned long translate_two_level(unsigned long virtual_addr,
                                   PageDirectory *pd,
                                   int page_size) {
    unsigned long addr = virtual_addr;
    
    unsigned int pd_index = (addr >> 22) & 0x3FF;
    unsigned int pt_index = (addr >> 12) & 0x3FF;
    unsigned int offset = addr & 0xFFF;
    
    PageTable *pt = (PageTable*)(pd->entries[pd_index].frame * page_size);
    
    if (!pd->entries[pd_index].present || !pt->entries[pt_index].present) {
        return 0;
    }
    
    unsigned long physical_frame = pt->entries[pt_index].frame;
    return (physical_frame * page_size) + offset;
}
```

### 3.3.3 翻頁緩衝區（TLB）

TLB是硬體快取，用於加速虛擬位址到實體位址的轉換。

```c
#define TLB_SIZE 64

typedef struct {
    unsigned int valid;
    unsigned int tag;
    unsigned int frame;
    int pid;
} TLBEntry;

TLBEntry tlb[TLB_SIZE];
int tlb_counter = 0;

unsigned long tlb_lookup(unsigned int pid, unsigned long virtual_addr) {
    for (int i = 0; i < TLB_SIZE; i++) {
        if (tlb[i].valid && 
            tlb[i].pid == pid && 
            tlb[i].tag == (virtual_addr >> 12)) {
            unsigned long offset = virtual_addr & 0xFFF;
            return (tlb[i].frame << 12) | offset;
        }
    }
    return 0;
}

void tlb_insert(unsigned int pid, unsigned long virtual_addr, unsigned long frame) {
    int index = tlb_counter % TLB_SIZE;
    tlb[index].valid = 1;
    tlb[index].pid = pid;
    tlb[index].tag = virtual_addr >> 12;
    tlb[index].frame = frame;
    tlb_counter++;
}

void tlb_flush(void) {
    for (int i = 0; i < TLB_SIZE; i++) {
        tlb[i].valid = 0;
    }
}

void tlb_flush_pid(unsigned int pid) {
    for (int i = 0; i < TLB_SIZE; i++) {
        if (tlb[i].pid == pid) {
            tlb[i].valid = 0;
        }
    }
}
```

## 3.4 分段機制

分段是另一種記憶體管理技術，提供更大的彈性。

### 3.4.1 分段表

```c
typedef struct {
    unsigned int base;       // 區段基底位址
    unsigned int limit;      // 區段大小
    unsigned int present : 1;
    unsigned int read : 1;
    unsigned int write : 1;
    unsigned int execute : 1;
    unsigned int privilege : 2;
} SegmentDescriptor;

typedef struct {
    SegmentDescriptor segments[6];
} GDT;

void init_gdt(GDT *gdt) {
    gdt->segments[0].present = 0;
    
    gdt->segments[1].base = 0x00000000;
    gdt->segments[1].limit = 0xFFFFF;
    gdt->segments[1].present = 1;
    gdt->segments[1].read = 1;
    gdt->segments[1].write = 1;
    gdt->segments[1].execute = 1;
    gdt->segments[1].privilege = 0;
    
    gdt->segments[2].base = 0x00000000;
    gdt->segments[2].limit = 0xFFFFF;
    gdt->segments[2].present = 1;
    gdt->segments[2].read = 1;
    gdt->segments[2].write = 1;
    gdt->segments[2].execute = 0;
    gdt->segments[2].privilege = 0;
}
```

### 3.4.2 段選擇子與位址轉換

```c
typedef struct {
    unsigned short index : 13;
    unsigned short ti : 1;
    unsigned short rpl : 2;
} SegmentSelector;

unsigned long translate_segmented(unsigned long virtual_addr,
                                   SegmentSelector selector,
                                   GDT *gdt) {
    int index = selector.index;
    SegmentDescriptor *desc = &gdt->segments[index];
    
    if (!desc->present) {
        fprintf(stderr, "Segment not present!\n");
        return 0;
    }
    
    if (virtual_addr > desc->limit) {
        fprintf(stderr, "Segmentation fault!\n");
        return 0;
    }
    
    return desc->base + virtual_addr;
}
```

## 3.5 動態記憶體配置

### 3.5.1 malloc 和 free 的實現

```c
typedef struct block_header {
    size_t size;
    int free;
    struct block_header *next;
} BlockHeader;

BlockHeader *head = NULL;
size_t total_size = 0;

#define HEADER_SIZE sizeof(BlockHeader)

void *malloc(size_t size) {
    if (size == 0) return NULL;
    
    BlockHeader *current = head;
    
    while (current) {
        if (current->free && current->size >= size) {
            current->free = 0;
            return (void*)(current + 1);
        }
        current = current->next;
    }
    
    BlockHeader *new_block = sbrk(sizeof(BlockHeader) + size);
    if (new_block == (void*)-1) return NULL;
    
    new_block->size = size;
    new_block->free = 0;
    new_block->next = NULL;
    
    if (head == NULL) {
        head = new_block;
    }
    
    return (void*)(new_block + 1);
}

void free(void *ptr) {
    if (ptr == NULL) return;
    
    BlockHeader *header = ((BlockHeader*)ptr) - 1;
    header->free = 1;
}
```

### 3.5.2 記憶體池配置器

```c
typedef struct memory_pool {
    void *memory;
    size_t total_size;
    size_t used_size;
    struct memory_pool *next;
} MemoryPool;

MemoryPool* pool_create(size_t size) {
    MemoryPool *pool = (MemoryPool*)malloc(sizeof(MemoryPool));
    pool->memory = malloc(size);
    pool->total_size = size;
    pool->used_size = 0;
    pool->next = NULL;
    return pool;
}

void* pool_alloc(MemoryPool *pool, size_t size) {
    if (pool->used_size + size > pool->total_size) {
        return NULL;
    }
    
    void *ptr = (char*)pool->memory + pool->used_size;
    pool->used_size += size;
    return ptr;
}

void pool_destroy(MemoryPool *pool) {
    free(pool->memory);
    free(pool);
}
```

## 3.6 記憶體對齊

記憶體對齊是確保資料結構位於特定邊界上的要求。

### 3.6.1 對齊要求

```c
#include <stdalign.h>

// 未對齊的結構
struct Unaligned {
    char a;
    int b;
    char c;
};
// 大小可能是 9 或 12，取決於編譯器

// 對齊的結構
struct Aligned {
    char a;
    char padding1[3];
    int b;
    char c;
    char padding2[3];
};
// 大小是 12

// 使用 #pragma pack
#pragma pack(push, 1)
struct Packed {
    char a;
    int b;
    char c;
};
#pragma pack(pop)
// 大小是 6
```

### 3.6.2 對齊函數

```c
size_t align_up(size_t size, size_t alignment) {
    return (size + alignment - 1) & ~(alignment - 1);
}

size_t align_down(size_t size, size_t alignment) {
    return size & ~(alignment - 1);
}

void* aligned_malloc(size_t size, size_t alignment) {
    void *ptr;
    size_t total = size + alignment - 1 + sizeof(void*);
    
    if (posix_memalign(&ptr, alignment, total) != 0) {
        return NULL;
    }
    
    void **aligned = (void**)((char*)ptr + sizeof(void*));
    aligned[-1] = ptr;
    
    return aligned;
}

void aligned_free(void *ptr) {
    if (ptr) {
        void **aligned = (void**)ptr;
        free(aligned[-1]);
    }
}
```

## 3.7 記憶體保護

### 3.7.1 保護機制

```c
#include <sys/mman.h>
#include <errno.h>

void* protected_alloc(size_t size) {
    void *ptr = mmap(NULL, size, 
                     PROT_READ | PROT_WRITE,
                     MAP_PRIVATE | MAP_ANONYMOUS,
                     -1, 0);
    
    if (ptr == MAP_FAILED) {
        return NULL;
    }
    
    return ptr;
}

int make_readonly(void *ptr, size_t size) {
    if (mprotect(ptr, size, PROT_READ) == -1) {
        return -1;
    }
    return 0;
}

int make_executable(void *ptr, size_t size) {
    if (mprotect(ptr, size, PROT_READ | PROT_EXEC) == -1) {
        return -1;
    }
    return 0;
}
```

### 3.7.2 記憶體洩漏檢測

```c
typedef struct allocation {
    void *ptr;
    size_t size;
    const char *file;
    int line;
    struct allocation *next;
} Allocation;

Allocation *allocations = NULL;

void* debug_malloc(size_t size, const char *file, int line) {
    void *ptr = malloc(size);
    
    Allocation *alloc = (Allocation*)malloc(sizeof(Allocation));
    alloc->ptr = ptr;
    alloc->size = size;
    alloc->file = file;
    alloc->line = line;
    alloc->next = allocations;
    allocations = alloc;
    
    return ptr;
}

void debug_free(void *ptr) {
    Allocation *prev = NULL;
    Allocation *current = allocations;
    
    while (current) {
        if (current->ptr == ptr) {
            if (prev) {
                prev->next = current->next;
            } else {
                allocations = current->next;
            }
            free(current);
            free(ptr);
            return;
        }
        prev = current;
        current = current->next;
    }
    
    fprintf(stderr, "Error: trying to free unallocated memory!\n");
}

void report_leaks(void) {
    Allocation *current = allocations;
    int count = 0;
    size_t total = 0;
    
    while (current) {
        fprintf(stderr, "Leak: %p (%zu bytes) at %s:%d\n",
                current->ptr, current->size, current->file, current->line);
        total += current->size;
        count++;
        current = current->next;
    }
    
    fprintf(stderr, "Total leaks: %d, %zu bytes\n", count, total);
}

#define malloc(size) debug_malloc(size, __FILE__, __LINE__)
#define free(ptr) debug_free(ptr)
```

## 3.8 交換空間管理

### 3.8.1 交換概念

交換（Swapping）將記憶體中的頁面移到磁碟，以釋放記憶體空間。

```c
typedef struct {
    unsigned int present : 1;
    unsigned int modified : 1;
    unsigned int referenced : 1;
    unsigned int swapped : 1;
    unsigned int frame : 20;
    unsigned int swap_location;
} SwapEntry;

typedef struct {
    unsigned long pages_written;
    unsigned long pages_read;
    unsigned long page_faults;
    unsigned long page_hits;
} SwapStats;

SwapStats swap_stats = {0, 0, 0, 0};

int swap_page_out(unsigned int frame, unsigned int swap_location) {
    FILE *swap_file = fopen("/tmp/swap", "r+b");
    if (!swap_file) return -1;
    
    fseek(swap_file, swap_location * PAGE_SIZE, SEEK_SET);
    
    void *phys_addr = (void*)(frame * PAGE_SIZE);
    fwrite(phys_addr, 1, PAGE_SIZE, swap_file);
    
    fclose(swap_file);
    swap_stats.pages_written++;
    
    return 0;
}

int swap_page_in(unsigned int swap_location, unsigned int frame) {
    FILE *swap_file = fopen("/tmp/swap", "rb");
    if (!swap_file) return -1;
    
    fseek(swap_file, swap_location * PAGE_SIZE, SEEK_SET);
    
    void *phys_addr = (void*)(frame * PAGE_SIZE);
    fread(phys_addr, 1, PAGE_SIZE, swap_file);
    
    fclose(swap_file);
    swap_stats.pages_read++;
    
    return 0;
}
```

### 3.8.2 頁面替換演算法

```c
typedef struct {
    unsigned int frame_number;
    unsigned int last_used;
    int referenced;
} FrameEntry;

FrameEntry *frame_table;
int frame_count;
int clock_hand = 0;

unsigned int clock_algorithm(void) {
    while (1) {
        FrameEntry *entry = &frame_table[clock_hand];
        
        if (!entry->referenced) {
            unsigned int victim = clock_hand;
            clock_hand = (clock_hand + 1) % frame_count;
            return victim;
        }
        
        entry->referenced = 0;
        clock_hand = (clock_hand + 1) % frame_count;
    }
}

unsigned int lru_algorithm(unsigned int *page_reference, int ref_count) {
    unsigned int oldest = 0;
    unsigned int min_time = page_reference[0];
    
    for (int i = 0; i < frame_count; i++) {
        if (frame_table[i].last_used < min_time) {
            min_time = frame_table[i].last_used;
            oldest = i;
        }
    }
    
    return oldest;
}

unsigned int fifo_algorithm(void) {
    unsigned int victim = clock_hand;
    clock_hand = (clock_hand + 1) % frame_count;
    return victim;
}
```
