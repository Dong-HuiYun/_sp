# 第四章：行程與執行序

## 4.1 行程的概念

行程（Process）是作業系統中的一個執行實體，代表了一個正在運行的程式。

### 4.1.1 行程的定義

行程不僅僅是程式的本身，而是程式的執行實例。它包含：

- 程式的程式碼
- 目前的活動
- 相關的記憶體內容
- 系統資源（如檔案描述符）
- 行程狀態

### 4.1.2 行程控制區塊（PCB）

行程控制區塊是作業系統用來儲存行程資訊的核心資料結構。

```c
typedef struct process_control_block {
    pid_t pid;                      // 行程ID
    pid_t parent_pid;               // 父行程ID
    process_state state;            // 行程狀態
    unsigned int priority;          // 行程優先級
    unsigned long start_time;       // 開始時間
    unsigned long cpu_time;         // 使用的CPU時間
    
    // 記憶體管理資訊
    unsigned long code_start;       // 程式碼起始位址
    unsigned long code_end;         // 程式碼結束位址
    unsigned long stack_pointer;    // 堆疊指標
    unsigned long base_pointer;     // 基底指標
    
    // 檔案相關資訊
    struct file *files[OPEN_MAX];   // 開啟的檔案
    int num_files;
    
    // 行程間通訊
    struct message_queue *msg_queue;
    struct semaphores *sems;
    
    // 排程資訊
    unsigned long nice_value;
    unsigned long counter;
    unsigned long priority_level;
    
    struct process_control_block *next;
    struct process_control_block *parent;
} PCB;
```

## 4.2 行程的創建與終止

### 4.2.1 Linux 中的行程創建

```c
#include <sys/types.h>
#include <unistd.h>
#include <stdio.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();
    
    if (pid < 0) {
        fprintf(stderr, "Fork failed!\n");
        return 1;
    } else if (pid == 0) {
        // 子行程
        printf("Child process: PID = %d, Parent PID = %d\n", 
               getpid(), getppid());
        
        char *args[] = {"./child_program", "arg1", "arg2", NULL};
        execvp("./child_program", args);
        
        perror("execvp failed");
        _exit(1);
    } else {
        // 父行程
        printf("Parent process: PID = %d, Child PID = %d\n", 
               getpid(), pid);
        
        int status;
        waitpid(pid, &status, 0);
        
        if (WIFEXITED(status)) {
            printf("Child exited with code %d\n", WEXITSTATUS(status));
        }
    }
    
    return 0;
}
```

## 4.3 執行序的概念

執行序（Thread）是行程內的執行單位，也稱為輕量級行程。

### 4.3.1 行程 vs 執行序

```

│           行程                     │
├────────────────────────────────────┤
│ ┌──────────┐ ┌──────────┐         │
│ │ 執行序1  │ │ 執行序2  │         │
│ ├──────────┤ ├──────────┤         │
│ │ 堆疊    │ │ 堆疊    │         │
│ ├──────────┤ ├──────────┤         │
```

## 4.4 POSIX 執行序

### 4.4.1 執行序的創建與終止

```c
#define _GNU_SOURCE
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

void* thread_function(void *arg) {
    int thread_id = *(int*)arg;
    
    printf("Thread %d: Starting\n", thread_id);
    
    for (int i = 0; i < 5; i++) {
        printf("Thread %d: iteration %d\n", thread_id, i);
        sleep(1);
    }
    
    printf("Thread %d: Finished\n", thread_id);
    
    int *result = (int*)malloc(sizeof(int));
    *result = thread_id * 10;
    
    return result;
}

int main() {
    pthread_t threads[3];
    int thread_args[3];
    
    for (int i = 0; i < 3; i++) {
        thread_args[i] = i + 1;
        
        int ret = pthread_create(&threads[i], NULL, 
                                  thread_function, 
                                  &thread_args[i]);
        
        if (ret != 0) {
            fprintf(stderr, "pthread_create failed!\n");
            return 1;
        }
    }
    
    for (int i = 0; i < 3; i++) {
        void *result;
        pthread_join(threads[i], &result);
        
        if (result) {
            printf("Thread %d returned %d\n", i + 1, *(int*)result);
            free(result);
        }
    }
    
    printf("Main: All threads completed\n");
    
    return 0;
}
```

## 4.5 執行序同步

### 4.5.1 互斥鎖

```c
#include <pthread.h>
#include <stdio.h>

pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
int counter = 0;

void* increment_counter(void *arg) {
    int iterations = *(int*)arg;
    
    for (int i = 0; i < iterations; i++) {
        pthread_mutex_lock(&mutex);
        
        int temp = counter;
        temp++;
        counter = temp;
        
        pthread_mutex_unlock(&mutex);
    }
    
    return NULL;
}

int main() {
    pthread_t threads[5];
    int iterations = 1000;
    
    for (int i = 0; i < 5; i++) {
        pthread_create(&threads[i], NULL, increment_counter, &iterations);
    }
    
    for (int i = 0; i < 5; i++) {
        pthread_join(threads[i], NULL);
    }
    
    printf("Final counter value: %d (expected: %d)\n", counter, 5 * iterations);
    
    pthread_mutex_destroy(&mutex);
    
    return 0;
}
```

### 4.5.2 讀寫鎖

```c
typedef struct {
    pthread_mutex_t mutex;
    pthread_cond_t read_cond;
    pthread_cond_t write_cond;
    int readers;
    int writers_waiting;
    int writer_active;
} ReadWriteLock;

void rwlock_init(ReadWriteLock *rwlock) {
    pthread_mutex_init(&rwlock->mutex, NULL);
    pthread_cond_init(&rwlock->read_cond, NULL);
    pthread_cond_init(&rwlock->write_cond, NULL);
    rwlock->readers = 0;
    rwlock->writers_waiting = 0;
    rwlock->writer_active = 0;
}

void rwlock_read_lock(ReadWriteLock *rwlock) {
    pthread_mutex_lock(&rwlock->mutex);
    
    while (rwlock->writer_active || rwlock->writers_waiting > 0) {
        pthread_cond_wait(&rwlock->read_cond, &rwlock->mutex);
    }
    
    rwlock->readers++;
    
    pthread_mutex_unlock(&rwlock->mutex);
}

void rwlock_read_unlock(ReadWriteLock *rwlock) {
    pthread_mutex_lock(&rwlock->mutex);
    
    rwlock->readers--;
    
    if (rwlock->readers == 0 && rwlock->writers_waiting > 0) {
        pthread_cond_signal(&rwlock->write_cond);
    }
    
    pthread_mutex_unlock(&rwlock->mutex);
}

void rwlock_write_lock(ReadWriteLock *rwlock) {
    pthread_mutex_lock(&rwlock->mutex);
    
    rwlock->writers_waiting++;
    
    while (rwlock->readers > 0 || rwlock->writer_active) {
        pthread_cond_wait(&rwlock->write_cond, &rwlock->mutex);
    }
    
    rwlock->writers_waiting--;
    rwlock->writer_active = 1;
    
    pthread_mutex_unlock(&rwlock->mutex);
}

void rwlock_write_unlock(ReadWriteLock *rwlock) {
    pthread_mutex_lock(&rwlock->mutex);
    
    rwlock->writer_active = 0;
    
    if (rwlock->writers_waiting > 0) {
        pthread_cond_signal(&rwlock->write_cond);
    } else {
        pthread_cond_broadcast(&rwlock->read_cond);
    }
    
    pthread_mutex_unlock(&rwlock->mutex);
}
```

### 4.5.3 條件變數

```c
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    pthread_mutex_t mutex;
    pthread_cond_t cond;
    int data_ready;
    int data;
} ProducerConsumer;

void pc_init(ProducerConsumer *pc) {
    pthread_mutex_init(&pc->mutex, NULL);
    pthread_cond_init(&pc->cond, NULL);
    pc->data_ready = 0;
    pc->data = 0;
}

void* producer(void *arg) {
    ProducerConsumer *pc = (ProducerConsumer*)arg;
    
    for (int i = 1; i <= 10; i++) {
        pthread_mutex_lock(&pc->mutex);
        
        while (pc->data_ready) {
            pthread_cond_wait(&pc->cond, &pc->mutex);
        }
        
        pc->data = i;
        pc->data_ready = 1;
        
        printf("Producer: produced %d\n", i);
        
        pthread_cond_signal(&pc->cond);
        pthread_mutex_unlock(&pc->mutex);
        
        usleep(100000);
    }
    
    return NULL;
}

void* consumer(void *arg) {
    ProducerConsumer *pc = (ProducerConsumer*)arg;
    
    for (int i = 0; i < 10; i++) {
        pthread_mutex_lock(&pc->mutex);
        
        while (!pc->data_ready) {
            pthread_cond_wait(&pc->cond, &pc->mutex);
        }
        
        int data = pc->data;
        pc->data_ready = 0;
        
        printf("Consumer: consumed %d\n", data);
        
        pthread_cond_signal(&pc->cond);
        pthread_mutex_unlock(&pc->mutex);
    }
    
    return NULL;
}
```

### 4.5.4 信號量

```c
#include <semaphore.h>
#include <pthread.h>
#include <stdio.h>

#define BUFFER_SIZE 10

typedef struct {
    int buffer[BUFFER_SIZE];
    int in;
    int out;
    sem_t empty;
    sem_t full;
    pthread_mutex_t mutex;
} CircularBuffer;

void cb_init(CircularBuffer *cb) {
    sem_init(&cb->empty, 0, BUFFER_SIZE);
    sem_init(&cb->full, 0, 0);
    pthread_mutex_init(&cb->mutex, NULL);
    cb->in = 0;
    cb->out = 0;
}

void cb_produce(CircularBuffer *cb, int item) {
    sem_wait(&cb->empty);
    pthread_mutex_lock(&cb->mutex);
    
    cb->buffer[cb->in] = item;
    cb->in = (cb->in + 1) % BUFFER_SIZE;
    
    printf("Produced: %d\n", item);
    
    pthread_mutex_unlock(&cb->mutex);
    sem_post(&cb->full);
}

int cb_consume(CircularBuffer *cb) {
    sem_wait(&cb->full);
    pthread_mutex_lock(&cb->mutex);
    
    int item = cb->buffer[cb->out];
    cb->out = (cb->out + 1) % BUFFER_SIZE;
    
    printf("Consumed: %d\n", item);
    
    pthread_mutex_unlock(&cb->mutex);
    sem_post(&cb->empty);
    
    return item;
}
```

## 4.7 行程排程

### 4.7.1 排程演算法

```c
typedef struct {
    int pid;
    int burst_time;
    int arrival_time;
    int priority;
    int remaining_time;
    int waiting_time;
    int turnaround_time;
    process_state state;
} Process;

void fcfs_schedule(Process *processes, int n) {
    int current_time = 0;
    
    for (int i = 0; i < n; i++) {
        if (current_time < processes[i].arrival_time) {
            current_time = processes[i].arrival_time;
        }
        
        processes[i].waiting_time = current_time - processes[i].arrival_time;
        current_time += processes[i].burst_time;
        processes[i].turnaround_time = current_time - processes[i].arrival_time;
    }
}

void sjf_schedule(Process *processes, int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (processes[j].burst_time < processes[i].burst_time) {
                Process temp = processes[i];
                processes[i] = processes[j];
                processes[j] = temp;
            }
        }
    }
    
    fcfs_schedule(processes, n);
}

void round_robin_schedule(Process *processes, int n, int quantum) {
    int current_time = 0;
    int completed = 0;
    int current_index = 0;
    
    while (completed < n) {
        if (processes[current_index].remaining_time > 0 &&
            processes[current_index].arrival_time <= current_time) {
            
            int exec_time = (processes[current_index].remaining_time > quantum) 
                           ? quantum 
                           : processes[current_index].remaining_time;
            
            processes[current_index].remaining_time -= exec_time;
            current_time += exec_time;
            
            if (processes[current_index].remaining_time == 0) {
                processes[current_index].turnaround_time = 
                    current_time - processes[current_index].arrival_time;
                processes[current_index].waiting_time = 
                    processes[current_index].turnaround_time - 
                    processes[current_index].burst_time;
                completed++;
            }
        }
        
        current_index = (current_index + 1) % n;
    }
}
```

### 4.7.2 多級反饋佇列

```c
#define NUM_QUEUES 3

typedef struct {
    Process *processes;
    int size;
    int front;
    int rear;
} Queue;

void enqueue(Queue *q, Process p) {
    if ((q->rear + 1) % q->size == q->front) {
        return;
    }
    q->processes[q->rear] = p;
    q->rear = (q->rear + 1) % q->size;
}

Process dequeue(Queue *q) {
    Process p = q->processes[q->front];
    q->front = (q->front + 1) % q->size;
    return p;
}

int is_empty(Queue *q) {
    return q->front == q->rear;
}

void mlfq_schedule(Process *processes, int n) {
    Queue queues[NUM_QUEUES];
    int time_quantum[NUM_QUEUES] = {4, 8, 16};
    
    for (int i = 0; i < NUM_QUEUES; i++) {
        queues[i].processes = (Process*)malloc(n * sizeof(Process));
        queues[i].size = n;
        queues[i].front = 0;
        queues[i].rear = 0;
    }
    
    int current_time = 0;
    int current_queue = 0;
    int completed = 0;
    
    while (completed < n) {
        for (int i = 0; i < n; i++) {
            if (processes[i].arrival_time == current_time) {
                enqueue(&queues[0], processes[i]);
            }
        }
        
        for (int q = 0; q < NUM_QUEUES; q++) {
            if (!is_empty(&queues[q])) {
                Process p = dequeue(&queues[q]);
                
                int exec_time = (p.remaining_time > time_quantum[q]) 
                               ? time_quantum[q] 
                               : p.remaining_time;
                
                p.remaining_time -= exec_time;
                current_time += exec_time;
                
                if (p.remaining_time == 0) {
                    completed++;
                } else {
                    int next_queue = (q + 1 < NUM_QUEUES) ? q + 1 : q;
                    enqueue(&queues[next_queue], p);
                }
                break;
            }
        }
    }
    
    for (int i = 0; i < NUM_QUEUES; i++) {
        free(queues[i].processes);
    }
}
```

2. 行程控制區塊的設計
3. 行程狀態與生命週期
4. 執行序的概念與優勢
5. POSIX 執行序程式設計
6. 執行序同步機制（互斥鎖、讀寫鎖、條件變數、信號量）
7. 行程間通訊方法（管道、訊息佇列、共享記憶體、訊號）
8. 行程排程演算法

## 4.9 習題

1. 解釋行程和執行序的區別。
2. 什麼是行程控制區塊？它包含哪些資訊？
3. 說明行程的各種狀態及其轉換。
4. 為什麼需要執行序同步？列舉常見的同步機制。
5. 比較互斥鎖和讀寫鎖的適用場景。
6. 說明生產者-消費者問題的解決方法。
7. 比較 FIFO、SJF 和 Round-Robin 排程演算法。
8. 實現一個使用管道進行父子行程通訊的程式。
