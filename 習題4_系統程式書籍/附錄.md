
### F.2 錯誤處理範本

```c
#include <stdio.h>
#include <stdlib.h>
#include <errno.h>
#include <string.h>

void error_exit(const char *message) {
    fprintf(stderr, "Error: %s\n", message);
    fprintf(stderr, "System error: %s\n", strerror(errno));
    exit(EXIT_FAILURE);
}

void *safe_malloc(size_t size) {
    void *ptr = malloc(size);
    if (ptr == NULL) {
        error_exit("Memory allocation failed");
    }
    return ptr;
}
```
